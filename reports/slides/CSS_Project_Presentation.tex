\begin{frame}{Motivation}
\begin{itemize}
  \item Minimal edits can change event structure without large lexical change.
  \item We test whether internal representations track these structural edits.
  \item Focused phenomena: role reversal and negation.
\end{itemize}

\begin{frame}{Research Questions}
\begin{block}{RQ1}
Do representation-shift metrics respond consistently to minimal structural edits across layers and models?
\end{block}
\begin{block}{RQ2}
Does matrix-norm (Frobenius) shift provide complementary signal beyond centroid cosine?
\end{block}
\begin{block}{RQ3}
How do probe selectivity and surprisal interact with representation-shift diagnostics?
\end{block}
\end{frame}

\begin{frame}{CSS Protocol}
\footnotesize
\[
\begin{aligned}
\text{Sentence Pair}
&\rightarrow
\text{Hidden States (layers 0--12)}
\rightarrow
\text{Shift Metrics}
\\
&\rightarrow
\text{Probes \& Surprisal}
\rightarrow
\text{Statistical Summary}
\end{aligned}
\]
\normalsize
\vspace{0.4em}
\begin{itemize}
  \item One paired comparison per item.
  \item All outputs computed per model and per layer.
  \item Fully config-driven and reproducible execution.
\end{itemize}
\end{frame}

\begin{frame}{Metric Intuition (What Each Metric Captures)}
\begin{itemize}
  \item \(\Delta_{\cos}\): direction change of sentence-level representation.
  \item \(\Delta_{\text{frob}}\): structural change in token-token interaction geometry.
  \item \(\Delta_{L2}\): absolute displacement magnitude in embedding space.
  \item \(\Delta_{\text{token}}\): average local token-level perturbation across aligned positions.
\end{itemize}
\vspace{0.3em}
\begin{block}{Interpretation Principle}
Large shifts indicate stronger representational response to a minimal counterfactual edit.
\end{block}
\end{frame}

\begin{frame}{Metric Definitions}
\small
\[
\Delta_{\cos}^{(l)} = 1 - \cos\!\big(\mu_l(s),\mu_l(s_{cf})\big)
\]
\[
\text{sim}_{\text{frob}}^{(l)} =
\frac{\lVert K(\widehat{A}_l,\widehat{B}_l)\rVert_F}
{\sqrt{\lVert K(\widehat{A}_l,\widehat{A}_l)\rVert_F\lVert K(\widehat{B}_l,\widehat{B}_l)\rVert_F + \varepsilon}}
,\quad
\Delta_{\text{frob}}^{(l)} = 1 - \text{sim}_{\text{frob}}^{(l)}
\]
\[
\Delta_{L2}^{(l)} = \lVert \mu_l(s)-\mu_l(s_{cf}) \rVert_2
\]
\[
\Delta_{\text{token}}^{(l)} =
\frac{1}{|M|}\sum_{(i,j)\in M}\left(1-\cos\!\big(h^{(l)}_{i},h^{(l)}_{j,cf}\big)\right)
\]
\normalsize
\end{frame}

\begin{frame}{Probes and Controls}
\begin{itemize}
  \item Linear probes by layer:
  role (\(agent/patient\)) and negation (\(0/1\)).
  \item Random-label baseline used for selectivity controls
  (\href{https://aclanthology.org/D19-1275/}{Hewitt \& Liang, 2019}).
  \item Selectivity:
  \[
  \text{Selectivity} = \text{Task Macro-F1} - \text{Control Macro-F1}
  \]
  \item Multi-seed evaluation for stability.
\end{itemize}
\end{frame}

\begin{frame}{Surprisal Signal}
\begin{itemize}
  \item Primary probabilistic signal is autoregressive GPT-2 surprisal.
  \item Formal definition:
  \[
  \text{surprisal}(t_i) = - \log P(t_i \mid t_1,\ldots,t_{i-1})
  \]
  \item Features used:
  total surprisal, average surprisal, and absolute counterfactual deltas.
  \item Key region coverage in this run: \(100\%\).
\end{itemize}
\end{frame}

\begin{frame}{Statistical Objectives}
\small
\textbf{D1:} layer-wise metric--surprisal correlations (Spearman primary, Pearson secondary).\\
\textbf{D2:} incremental value of \(\Delta_{\text{frob}}\) beyond \(\Delta_{\cos}\).\\
\textbf{D3:} layer-profile differences by model and phenomenon.\\
\textbf{D4:} probe selectivity stability across layers.
\vspace{0.4em}
\[
z(y) \sim z(\Delta_{\cos}) + z(\Delta_{\text{frob}}) + z(\text{length}) + z(\text{overlap})
\]
\normalsize
\vspace{0.3em}
\begin{itemize}
  \item Bootstrap confidence intervals.
  \item Benjamini--Hochberg FDR correction on layer-wise tests.
\end{itemize}
\end{frame}

\begin{frame}{Qualitative Audit: Why the Patterns Differ}
\small
\begin{table}
\centering
\begin{tabular}{lccc}
\toprule
\textbf{Phenomenon} & \textbf{Lexical Jaccard} & \textbf{Abs. length delta} & \textbf{Mean \(\Delta_{\text{frob}}\)} \\
\midrule
Negation & 0.4853 & 1.0013 & 0.0584 \\
Role reversal & 0.7973 & 0.0467 & 0.0197 \\
\bottomrule
\end{tabular}
\end{table}
\vspace{0.3em}
\begin{itemize}
  \item Negation pairs often combine a polarity cue with a predicate/category contrast.
  \item Role-reversal pairs preserve more lexical material and usually keep length fixed.
  \item Therefore, larger mean shift for negation does not contradict stronger role-reversal shift--surprisal correlations.
  \item Qualitative audit also flags generated-data artifacts, so claims remain dataset-level.
\end{itemize}
\end{frame}

\begin{frame}{Qualitative Case Studies}
\scriptsize
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.49\textwidth}
\textbf{High Frobenius / low surprisal}

\vspace{0.2em}
Negation example:

\emph{A hammer is an instrument.}

\(\rightarrow\) \emph{A hammer is not a dessert.}

\vspace{0.25em}
\(\Delta_{\text{frob}}=0.0763\), \(|\Delta\text{ avg surprisal}|=0.0037\)

\vspace{0.4em}
Interpretation: geometry moves strongly even when average predictive difficulty barely changes.
\end{column}
\begin{column}{0.49\textwidth}
\textbf{Low Frobenius / high surprisal}

\vspace{0.2em}
Role-reversal example:

\emph{The cashier counted which bills the robber had given.}

\(\rightarrow\) \emph{The cashier counted which robber the bills had given.}

\vspace{0.25em}
\(\Delta_{\text{frob}}=0.0138\), \(|\Delta\text{ avg surprisal}|=1.1860\)

\vspace{0.4em}
Interpretation: surprisal can change strongly even when averaged geometry moves less.
\end{column}
\end{columns}
\vspace{0.5em}
\centering
\small These dissociations motivate CSS as a multi-diagnostic framework.
\end{frame}
